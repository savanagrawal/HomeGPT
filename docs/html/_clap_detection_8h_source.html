112 // Compute the iterative in-place FFT of a vector
113 static void fft(std::vector<std::complex<double>>& a, bool inverse) {
114 int n = a.size();
115 int log_n = std::log2(n);
116
117 // Bit-reverse permutation
118 for (int i = 0; i < n; i++) {
119 int j = reverse_bits(i, log_n);
120 if (i < j) {
121 std::swap(a[i], a[j]);
122 }
123 }
124
125 // Iterative FFT
126 for (int len = 2; len <= n; len <<= 1) {
127 double angle = 2 * M_PI / len * (inverse ? -1 : 1);
128 std::complex<double> wn(std::cos(angle), std::sin(angle));
129 for (int i = 0; i < n; i += len) {
130 std::complex<double> w(1);
131 for (int j = 0; j < len / 2; j++) {
132 std::complex<double> u = a[i + j];
133 std::complex<double> v = a[i + j + len / 2] * w;
134 a[i + j] = u + v;
135 a[i + j + len / 2] = u - v;
136 w *= wn;
137 }
138 }
139 }
140
141 // If inverse FFT, divide by n
142 if (inverse) {
143 for (auto& x : a) {
144 x /= n;
145 }
146 }
147 }

149 //Compute the circular convolution of two vectors using the DFT
150 static std::vector<double> convolve(const std::vector<double>& a, const std::vector<double>& b) {
151 int n = a.size();
152 int m = b.size();
153 int l = n + m - 1;
154 int p = 1;
155 while (p < l) {
156 p <<= 1;
157 }
158 std::vector<std::complex<double>> pa(p), pb(p), pc(p);
159 std::copy(a.begin(), a.end(), pa.begin());
160 std::copy(b.begin(), b.end(), pb.begin());
161 fft(pa, false);
162 fft(pb, false);
163 std::transform(pa.begin(), pa.end(), pb.begin(), pc.begin(),
164 [](const std::complex<double>& a, const std::complex<double>& b) { return a * b; });
165 fft(pc, true);
166 std::vector<double> result(l);
167 // Use std::transform to compute the real part of each complex number
168 std::transform(pc.begin(), pc.begin() + l, result.begin(),
169 [](const std::complex<double>& cplx) { return std::real(cplx) * std::real(cplx); });
170
171 return result;
172 }
