189 int result = 0;
190 for (int i = 0; i < bit_length; i++) {
191 result = (result << 1) | (num & 1);

197 // Compute the iterative in-place FFT of a vector
198 static void fft(std::vector<std::complex<double>>& a, bool inverse) {
199 int n = a.size();
200 int log_n = std::log2(n);
201
202 // Bit-reverse permutation
203 for (int i = 0; i < n; i++) {
204 int j = reverse_bits(i, log_n);
205 if (i < j) {
206 std::swap(a[i], a[j]);
207 }
208 }
209
210 // Iterative FFT
211 for (int len = 2; len <= n; len <<= 1) {
212 double angle = 2 * M_PI / len * (inverse ? -1 : 1);
213 std::complex<double> wn(std::cos(angle), std::sin(angle));
214 for (int i = 0; i < n; i += len) {
215 std::complex<double> w(1);
216 for (int j = 0; j < len / 2; j++) {
217 std::complex<double> u = a[i + j];
218 std::complex<double> v = a[i + j + len / 2] * w;
219 a[i + j] = u + v;
220 a[i + j + len / 2] = u - v;
221 w *= wn;
222 }
223 }
224 }
225
226 // If inverse FFT, divide by n
227 if (inverse) {
228 for (auto& x : a) {
229 x /= n;
230 }
231 }
232 }

234 //Compute the circular convolution of two vectors using the DFT
235 static std::vector<double> convolve(const std::vector<double>& a, const std::vector<double>& b) {
236 int n = a.size();
237 int m = b.size();
238 int l = n + m - 1;
239 int p = 1;
240 while (p < l) {
241 p <<= 1;
242 }
243 std::vector<std::complex<double>> pa(p), pb(p), pc(p);
244 std::copy(a.begin(), a.end(), pa.begin());
245 std::copy(b.begin(), b.end(), pb.begin());
246 fft(pa, false);
247 fft(pb, false);
248 std::transform(pa.begin(), pa.end(), pb.begin(), pc.begin(),
249 [](const std::complex<double>& a, const std::complex<double>& b) { return a * b; });
250 fft(pc, true);
251 std::vector<double> result(l);
252 // Use std::transform to compute the real part of each complex number
253 std::transform(pc.begin(), pc.begin() + l, result.begin(),
254 [](const std::complex<double>& cplx) { return std::real(cplx) * std::real(cplx); });
255
256 return result;
257 }
